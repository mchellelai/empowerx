    <script>

        // Main script for chatbot logic

        document.addEventListener('DOMContentLoaded', () => {


            // DOM elements

            const chatWindow = document.getElementById('chat-window');

            const buttonArea = document.getElementById('button-area');

            const textInputContainer = document.getElementById('text-input-container');

            const fileUploadContainer = document.getElementById('file-upload-container');

            const userInput = document.getElementById('user-input');

            const sendButton = document.getElementById('send-button');

            const photoUploadInput = document.getElementById('photo-upload');

            const uploadButton = document.getElementById('upload-button');


            // Conversation state and data

            let conversationState = 0;

            let currentMedications = [];


            // Add a message to the chat window

            const addMessage = (text, isBot = true, isHTML = false) => {

                const messageBubble = document.createElement('div');

                messageBubble.classList.add('message-bubble', isBot ? 'bot-message' : 'user-message');

                if (isHTML) {

                    messageBubble.innerHTML = text;

                } else {

                    messageBubble.textContent = text;

                }

                chatWindow.appendChild(messageBubble);

                // Auto-scroll to the bottom

                chatWindow.scrollTop = chatWindow.scrollHeight;

            };


            // Add a photo to the chat window

            const addPhoto = (file) => {

                const reader = new FileReader();

                reader.onload = (e) => {

                    const messageBubble = document.createElement('div');

                    messageBubble.classList.add('message-bubble', 'user-message');

                    const img = document.createElement('img');

                    img.src = e.target.result;

                    img.alt = 'Uploaded photo';

                    img.classList.add('uploaded-image');

                    messageBubble.appendChild(img);

                    chatWindow.appendChild(messageBubble);

                    chatWindow.scrollTop = chatWindow.scrollHeight;

                };

                reader.readAsDataURL(file);

            };


            // Display dynamic buttons based on state

            const showButtons = (buttons) => {

                buttonArea.innerHTML = ''; // Clear previous buttons

                buttons.forEach(btn => {

                    const button = document.createElement('button');

                    button.textContent = btn.text;

                    button.classList.add('px-4', 'py-2', 'rounded-full', 'font-medium', 'bg-gray-200', 'text-gray-800', 'hover:bg-gray-300', 'transition-colors', 'duration-200');

                    button.addEventListener('click', () => btn.action(btn.text));

                    buttonArea.appendChild(button);

                });

            };


            // Main chatbot logic based on conversation state

            const handleMessage = (message) => {

                switch (conversationState) {

                    case 0:

                        addMessage('Hello there! I am Ember from EmpoweRx, here to help you update your medication list!');

                        addMessage('First, let\'s retrieve your medication list from HealthHub!');

                        showButtons([

                          { text: 'Retrieve Medication List', action: (text) => {

                              addMessage(text, false);

                              conversationState = 1;

                              handleMessage(text);

                          } },

                          { text: 'Start a New List', action: (text) => {

                              addMessage(text, false);

                              // Clear conversation and reset state

                              chatWindow.innerHTML = '';

                              conversationState = 2; // Move directly to adding new medications

                              currentMedications = [];

                              handleMessage('Enter manually'); // Automatically jump to manual entry

                              textInputContainer.style.display = 'flex';

                              fileUploadContainer.style.display = 'none';

                          } }

                        ]);

                        break;

                    

                    case 1:

                        addMessage('This is your medication list on HealthHub:');

                        const list1 = '1. Amlodipine 5mg tablet (Zynor) - 1 tablet every morning (supplied 90 tablets from Tan Tock Seng Hospital on 1 Jun 2023)';

                        addMessage(list1);

                        currentMedications.push(list1);

                        addMessage('Are you taking your medications according to these instructions?');

                        showButtons([{ text: 'Yes', action: (text) => {

                            addMessage(text, false);

                            conversationState = 2;

                            handleMessage(text);

                        } }]);

                        break;

                    

                    case 2:

                        addMessage('Do you have any other medications/supplements to add?');

                        showButtons([

                            { text: 'Take photo', action: (text) => {

                                addMessage(text, false);

                                conversationState = 3;

                                handleMessage(text);

                            }},

                            { text: 'Enter manually', action: (text) => {

                                addMessage(text, false);

                                conversationState = 3;

                                handleMessage(text);

                            }}

                        ]);

                        break;

                    

                    case 3:

                        if (message === 'Take photo') {

                            addMessage('Please upload a photo of your medication label. Ensure it is clearly visible.');

                            buttonArea.innerHTML = ''; // Clear buttons

                            textInputContainer.style.display = 'none';

                            fileUploadContainer.style.display = 'flex';

                        } else if (message === 'Enter manually') {

                            addMessage('Please enter the name of the medication or supplement you want to add.');

                            buttonArea.innerHTML = ''; // Clear buttons

                            textInputContainer.style.display = 'flex';

                            fileUploadContainer.style.display = 'none';

                        } else {

                            // User has entered a manual medication

                            const newMedication = message;

                            addMessage(newMedication, false);

                            addMessage(`Based on your input, you have added: 1. ${newMedication} Please reply Yes to confirm.`);

                            showButtons([{ text: 'Yes', action: (text) => {

                                addMessage(text, false);

                                conversationState = 4;

                                handleMessage(text);

                            } }]);

                            currentMedications.push(newMedication);

                        }

                        break;


                    case 4:

                        if (message === 'Yes') {

                            addMessage(`I see you have entered "${currentMedications[1].split(' ')[0]}". Did you mean "Dapagliflozin", a medication usually used for diabetes?`);

                            showButtons([{ text: 'Yes', action: (text) => {

                                addMessage(text, false);

                                conversationState = 5;

                                handleMessage(text);

                            } }]);

                        } else if (message === 'Dapagliflozin') { // Handling mock photo upload

                            addMessage(`I found "${message}" in your photo.`);

                            showButtons([{ text: 'Yes', action: (text) => {

                                addMessage(text, false);

                                conversationState = 5;

                                handleMessage(text);

                            } }]);

                        }

                        break;


                    case 5:

                        if (message === 'Yes') {

                            addMessage('What is the strength of the tablet you are taking? The tablets usually come in 5mg and 10mg.');

                            textInputContainer.style.display = 'flex';

                            fileUploadContainer.style.display = 'none';

                            buttonArea.innerHTML = '';

                            conversationState = 6;

                        }

                        break;

                        

                    case 6:

                        // User enters the strength

                        const strength = message;

                        addMessage(strength, false);

                        

                        // Final combined list

                        const combinedList = currentMedications.map((item, index) => {

                            if (index === 0) {

                                return `1. Amlodipine 5mg tablet (Zynor) - 1 tablet every morning (supplied 90 tablets from Tan Tock Seng Hospital on 1 Jun 2023)`;

                            } else {

                                // Assume manual entry was Dapagliflozin

                                const parsedMed = item;

                                return `2. Dapagliflozin ${strength} tablet once daily (from JB Pharmacy)`;

                            }

                        }).join('<br>');

                        

                        addMessage('Great! This is your medication list so far:');

                        addMessage(`<ul class="message-list">${combinedList}</ul>`, true, true);

                        addMessage('Is this correct?');

                        showButtons([{ text: 'Yes', action: (text) => {

                            addMessage(text, false);

                            conversationState = 7;

                            handleMessage(text);

                        } }, { text: 'No', action: (text) => {

                            addMessage(text, false);

                            conversationState = 7;

                            handleMessage(text);

                        } }]);

                        break;


                    case 7:

                        if (message === 'Yes') {

                            addMessage('Great. Do you need to add more?');

                            showButtons([{ text: 'Yes', action: (text) => {

                                addMessage(text, false);

                                conversationState = 8;

                                handleMessage(text);

                            }}, { text: 'No', action: (text) => {

                                addMessage(text, false);

                                conversationState = 8;

                                handleMessage(text);

                            }}]);

                        } else {

                            // Reset conversation if user says no

                            addMessage('No problem! Let\'s start over. What would you like to do?');

                            conversationState = 0;

                            handleMessage('start');

                        }

                        break;

                    

                    case 8:

                        if (message === 'No') {

                            addMessage('Great. I will proceed to save this list. Your healthcare provider will now be able to access your latest medication list from EmpoweRx.');

                            buttonArea.innerHTML = '';

                            textInputContainer.style.display = 'none';

                            fileUploadContainer.style.display = 'none';


                            // Add the restart button at the end

                            showButtons([{ text: 'Restart Conversation', action: () => {

                                // Clear conversation and reset state

                                chatWindow.innerHTML = '';

                                conversationState = 0;

                                currentMedications = [];

                                handleMessage('start');

                                textInputContainer.style.display = 'flex';

                                fileUploadContainer.style.display = 'none';

                            }}]);


                        } else {

                            // Loop back to adding more medications

                            conversationState = 3;

                            handleMessage('Enter manually');

                        }

                        break;

                    default:

                        // Reset if an unexpected state is reached

                        conversationState = 0;

                        handleMessage('start');

                        break;

                }

            };


            // Event listener for the "Send" button

            sendButton.addEventListener('click', () => {

                const message = userInput.value.trim();

                if (message) {

                    handleMessage(message);

                    userInput.value = ''; // Clear input field

                }

            });


            // Allow sending by pressing Enter key

            userInput.addEventListener('keypress', (e) => {

                if (e.key === 'Enter') {

                    sendButton.click();

                }

            });


            // Event listener for the "Upload Photo" button

            uploadButton.addEventListener('click', () => {

                const file = photoUploadInput.files[0];

                if (file) {

                    addPhoto(file); // Display the photo

                    addMessage(`Uploading photo of ${file.name}...`);

                    // Simulate processing and finding Dapagliflozin

                    currentMedications.push('Dapagliflozin');

                    conversationState = 4;

                    setTimeout(() => {

                        handleMessage('Dapagliflozin');

                    }, 1000); // Wait 1 second to simulate processing

                } else {

                    addMessage('Please select a photo to upload.', true);

                }

            });


            // Start the conversation
